]{geometry}
\usepackage{luacode}

\usepackage{scalefnt}
\usepackage{graphicx}

\setlength{\parindent}{0pt}

\begin{document}
\thispagestyle{empty}

\begin{luacode*}
local gridlib = require("gridlib")

-- Configuration
cfg = {
  start_year = 2004,       -- first year column
  num_years = 22,          -- how many years (columns)
}

-- Generate labels
local top_labels = gridlib.number_range(cfg.start_year, cfg.num_years)
local left_labels = gridlib.month_names

-- Get grid dimensions from TeX (set by geometry package)
local paper_width, paper_height, grid_width, grid_height = gridlib.get_page_dimensions()

-- Draw the grid
gridlib.draw_grid(grid_width, grid_height, top_labels, left_labels, {
  top_font_size = [[\LARGE]],
  left_font_size = [[\LARGE]],
  left_margin = 0.6,
})

-- Print the pdfcrop command for test printing
gridlib.print_crop_command("yeargrid.pdf", paper_width, paper_height)
\end{luacode*}

\end{document}
