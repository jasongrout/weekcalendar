\documentclass{article}

\usepackage[paperwidth=60in, paperheight=36in, margin=1in,
% Use showframe to show the margin space
% showframe
]{geometry}
\usepackage{luacode}

\usepackage{scalefnt}
\usepackage{graphicx}

\setlength{\parindent}{0pt}

\begin{document}
\thispagestyle{empty}

\begin{luacode*}
local gridlib = require("gridlib")

-- Configuration
cfg = {
  paper_width = 60,        -- in inches: keep in sync with the geometry command above
  paper_height = 36,       -- in inches: keep in sync with the geometry command above
  margin = 1,              -- in inches: keep in sync with the geometry command above
  start_year = 2026,       -- first year column
  num_years = 30,          -- how many years (columns)
}

-- Generate labels
local top_labels = gridlib.number_range(cfg.start_year, cfg.num_years)
local left_labels = gridlib.month_names

-- Calculate grid dimensions from page size
local grid_width = cfg.paper_width - 2 * cfg.margin
local grid_height = cfg.paper_height - 2 * cfg.margin

-- Draw the grid
gridlib.draw_grid(grid_width, grid_height, top_labels, left_labels, {
  top_font_size = [[\LARGE]],
  left_font_size = [[\LARGE]],
  left_margin = 0.6,
})
\end{luacode*}

\end{document}
